+async function icFetchOneSource(dbId) {
+  await getToken();
+  const btn = document.getElementById(`ic-fetch-btn-${dbId}`);
+  if (btn) { btn.disabled = true; btn.textContent = '…'; }
+  try {
+    const r = await fetch(`/api/ic/sources/${dbId}/fetch`, {method:'POST', headers: icAuthHeaders()});
+    const d = await r.json();
+    if (btn) { btn.disabled = false; btn.textContent = d.ok ? '✓' : '✗'; }
+    icLoadSources(); icLoadNewsCache();
+  } catch(e) {
+    if (btn) { btn.disabled = false; btn.textContent = '✗'; }
+  }
+}

+async function icLoadNewsCache() {
+  const wrap = document.getElementById('ic-news-cache-wrap');
+  if (!wrap) return;
+  try {
+    const r = await fetch('/api/ic/sources/news');
+    const data = await r.json();
+    if (!data.length) {
+      wrap.innerHTML = '<div class="ic-empty-msg" style="font-size:12px">尚無快取內容</div>';
+      return;
+    }
+    wrap.innerHTML = data.map(n => `<div class="ic-card" style="margin-bottom:8px">
+      <div style="display:flex;align-items:center;gap:8px;margin-bottom:6px">
+        <span class="ic-src-badge ic-src-user">◈ ${n.name}</span>
+        <span style="font-size:10px;color:var(--text2)">${n.market} · ${n.type}</span>
+        <span style="font-size:10px;color:var(--text2);margin-left:auto">${(n.fetched_at||'').slice(0,16)}</span>
+      </div>
+      <div style="font-size:12px;color:var(--text2);white-space:pre-wrap;line-height:1.5;background:var(--bg);border-radius:4px;padding:7px;max-height:110px;overflow-y:auto">${icEscHtml(n.content||'')}</div>
+    </div>`).join('');
+  } catch(e) {
+    wrap.innerHTML = '<div class="ic-empty-msg">載入失敗</div>';
+  }
+}

+async function icAddSource() {
+  await getToken();
+  const name = document.getElementById('ic-src-name')?.value.trim() || '';
+  if (!name) { alert('請填寫名稱'); return; }
+  try {
+    await fetch('/api/ic/sources', {
+      method: 'POST', headers: icAuthHeaders(),
+      body: JSON.stringify({
+        name,
+        url:         document.getElementById('ic-src-url')?.value.trim() || '',
+        type:        document.getElementById('ic-src-type')?.value || 'HTML',
+        market:      document.getElementById('ic-src-market')?.value || 'ALL',
+        reliability: document.getElementById('ic-src-reliability')?.value || 'reference',
+        description: document.getElementById('ic-src-desc')?.value.trim() || '',
+      }),
+    });
+    ['ic-src-name','ic-src-url','ic-src-desc'].forEach(id => {
+      const el = document.getElementById(id);
+      if (el) el.value = '';
+    });
+    const msg = document.getElementById('ic-src-msg');
+    if (msg) { msg.style.display = ''; setTimeout(() => msg.style.display='none', 2000); }
+    icLoadSources();
+  } catch(e) { alert('新增失敗'); }
+}

+async function icDeleteSource(dbId) {
+  if (!confirm('確定刪除此來源？')) return;
+  await getToken();
+  try {
+    await fetch(`/api/ic/sources/${dbId}`, { method: 'DELETE', headers: icAuthHeaders() });
+    icLoadSources();
+  } catch(e) { alert('刪除失敗'); }
+}
